## پیاده‌سازی در Python

```python
import requests

API_KEY = "YourApiKey"

# --- داده لحظه‌ای همه نمادها ---
def get_all_symbols(market_type=1):
    url = f"https://Api.BrsApi.ir/Tsetmc/AllSymbols.php?key={API_KEY}&type={market_type}"
    response = requests.get(url)
    return response.json()

# --- داده تاریخی یک نماد ---
def get_history(symbol):
    url = f"https://Api.BrsApi.ir/Tsetmc/History.php?key={API_KEY}&type=0&l18={symbol}"
    response = requests.get(url)
    return response.json()

# --- کندل تاریخی تعدیل‌شده ---
def get_candlestick_adjusted(symbol):
    url = f"https://Api.BrsApi.ir/Tsetmc/Candlestick.php?key={API_KEY}&type=3&l18={symbol}"
    response = requests.get(url)
    return response.json()

# --- کندل لحظه‌ای (روز جاری) ---
def get_candlestick_realtime(symbol):
    url = f"https://Api.BrsApi.ir/Tsetmc/Candlestick.php?key={API_KEY}&type=1&l18={symbol}"
    response = requests.get(url)
    return response.json()

# نمونه استفاده
if __name__ == "__main__":
    data = get_all_symbols()
    print(data)
    
    history = get_history("فملی")
```
